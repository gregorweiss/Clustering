126  void Node::set_pos(int x, int y) {
127  this->pos_x = x;
128  this->pos_y = y;
129  std::vector<int> width_children;
130  int total_width = 0;
131  for (auto& id_child: children) {
132  width_children.push_back(id_child.second.subtree_width());
133  total_width += id_child.second.subtree_width();
134  }
135  // set pos for every child recursively,
136  // regarding proper segmentation of horizontal space.
137  int cur_x = (x-0.5*total_width);
138  for (auto& id_child: children) {
139  int stw = id_child.second.subtree_width();
140  id_child.second.set_pos(cur_x + 0.5*stw, y + VERTICAL_SPACING);
141  cur_x += stw;
142  }
143  }

145  int Node::subtree_width() {
146  // check for backtracked value
147  // and compute if not existing.
148  if ( ! this->_subtree_width) {
149  int self_width = 10 + 2*HORIZONTAL_SPACING; //TODO get from size
150  if (children.empty()) {
151  this->_subtree_width = self_width;
152  } else {
153  int sum = 0;
154  for (auto& id_child: children) {
155  sum += id_child.second.subtree_width();
156  }
157  if (sum > self_width) {
158  this->_subtree_width = sum;
159  } else {
160  this->_subtree_width = self_width;
161  }
162  }
163  }
164  return this->_subtree_width;
165  }

223  std::size_t src = from_to.first;
224  std::size_t target = from_to.second;
225  not_leaves.insert(target);
226  if (not_leaves.count(src)) {
227  leaves.erase(src);
228  } else {
229  leaves.insert(src);
230  }
231  }
232  for (std::size_t leaf: leaves) {
233  ofs << leaf << "\n";
234  }
235  }

253  std::vector<std::size_t> traj(n_rows);
254  const float prec = d_step / 10.0f;
255  for (float d=d_min; ! fuzzy_equal(d, d_max+d_step, prec); d += d_step) {

257  Clustering::Tools::stringprintf(remapped_name, d));
258  for (std::size_t i=0; i < n_rows; ++i) {
259  if (leaves.count(cl_now[i])) {
260  traj[i] = cl_now[i];
261  }
262  }
263  }
264  for (std::size_t i: traj) {
265  ofs << i << "\n";

270  void
271  save_network_to_html(std::string fname,
272  std::map<std::size_t, std::size_t> network,
273  std::map<std::size_t, float> free_energies,
274  std::map<std::size_t, std::size_t> pops) {
275  Clustering::logger(std::cout) << "computing network visualization" << std::endl;
276  // set (global) values for min/max of free energies and populations
277  FE_MAX = std::max_element(free_energies.begin(),
278  free_energies.end(),
279  [](std::pair<std::size_t, float> fe1,
280  std::pair<std::size_t, float> fe2) -> bool {
281  return fe1.second < fe2.second;
282  })->second;
283  FE_MIN = std::min_element(free_energies.begin(),
284  free_energies.end(),
285  [](std::pair<std::size_t, float> fe1,
286  std::pair<std::size_t, float> fe2) -> bool {
287  return fe1.second < fe2.second;
288  })->second;
289  POP_MAX = std::max_element(pops.begin(),
290  pops.end(),
291  [](std::pair<std::size_t, std::size_t> p1,
292  std::pair<std::size_t, std::size_t> p2) -> bool {
293  return p1.second < p2.second;
294  })->second;
295  POP_MIN = std::min_element(pops.begin(),
296  pops.end(),
297  [](std::pair<std::size_t, std::size_t> p1,
298  std::pair<std::size_t, std::size_t> p2) -> bool {
299  return p1.second < p2.second;
300  })->second;
301  // build trees from given network with respective 'root' on top and at highest FE.
302  // may be multiple trees because there may be multiple nodes that have max FE.
303  Node fake_root;
304  std::size_t network_size = network.size();
305  std::size_t i_frame = 0;
306  for (auto from_to: network) {
307  std::cerr << ++i_frame << " / " << network_size << "\n";
308 
309  std::size_t i_from = from_to.first;
310  std::size_t i_to = from_to.second;
311 
312  Node* parent_to = fake_root.find_parent_of(i_to);
313  if ( ! parent_to) {
314  fake_root.children[i_to] = {i_to, free_energies[i_to], pops[i_to]};
315  parent_to = &fake_root;
316  }
317  Node* parent_from = fake_root.find_parent_of(i_from);
318  if (parent_from) {
319  // move existing node to its right place in the tree
320  parent_to->children[i_to].children[i_from] = parent_from->children[i_from];
321  parent_from->children.erase(i_from);
322  } else {
323  // create child at proper place in tree
324  parent_to->children[i_to].children[i_from] = {i_from, free_energies[i_from], pops[i_from]};
325  }
326  }
327  // write header
328  std::ofstream ofs(fname);
329  if (ofs.fail()) {
330  std::cerr << "error: cannot open file '" << fname << "' for writing." << std::endl;
331  exit(EXIT_FAILURE);
332  } else {
333  float LOG_POP_MIN, LOG_POP_MAX;
334  if (POP_MIN <= 0) {
335  LOG_POP_MIN = 0.0f;
336  } else {
337  LOG_POP_MIN = log(POP_MIN);
338  }
339  if (POP_MAX <= 0) {
340  LOG_POP_MAX = 0.0f;
341  } else {
342  LOG_POP_MAX = log(POP_MAX);
343  }
344  ofs << Clustering::Network::viewer_header
345 
346  << "style: cytoscape.stylesheet().selector('node').css({"
347  << Clustering::Tools::stringprintf("'width': 'mapData(logpop, %0.2f, %0.2f, 5, 30)',", LOG_POP_MIN, LOG_POP_MAX)
348  << Clustering::Tools::stringprintf("'height': 'mapData(logpop, %0.2f, %0.2f, 5, 30)',", LOG_POP_MIN, LOG_POP_MAX)
349  << Clustering::Tools::stringprintf("'background-color': 'mapData(fe, %f, %f, blue, red)'})", FE_MIN, FE_MAX)
350 
351  << ".selector('edge').css({'opacity': '1.0', 'width': '5', 'target-arrow-shape': 'triangle'})"
352  << ".selector(':selected').css({'content': 'data(info)', 'font-size': 24, 'color': '#00ff00'})"
353 
354  << ", elements: [\n";
355  fake_root.set_pos(0, 0);
356  fake_root.print_subtree(ofs);
357  ofs << "]";
358  ofs << Clustering::Network::viewer_footer;
359  }

367  void
368  main(boost::program_options::variables_map args) {
369  using namespace Clustering::Tools;
370  // setup two threads parallel read/write
371  omp_set_num_threads(2);
372  // setup general flags / options
373  Clustering::verbose = args["verbose"].as<bool>();
374 
375  float d_min = args["min"].as<float>();
376  float d_max = args["max"].as<float>();
377  float d_step = args["step"].as<float>();
378  std::string basename = args["basename"].as<std::string>();
379  std::string remapped_name = "remapped_" + basename;
380  std::size_t minpop = args["minpop"].as<std::size_t>();
381 
382  std::map<std::size_t, std::size_t> network;
383  std::map<std::size_t, std::size_t> pops;
384  std::map<std::size_t, float> free_energies;
385 
386  std::vector<std::size_t> cl_next = read_clustered_trajectory(stringprintf(basename, d_min));
387  std::vector<std::size_t> cl_now;
388  std::size_t max_id;
389  std::size_t n_rows = cl_next.size();
390  // re-map states to give every state a unique id.
391  // this is nevessary, since every initially clustered trajectory
392  // at different thresholds uses the same ids starting with 0.
393  const float prec = d_step / 10.0f;
394  for (float d=d_min; ! fuzzy_equal(d, d_max, prec); d += d_step) {
395  Clustering::logger(std::cout) << "free energy level: " << stringprintf("%0.2f", d) << std::endl;
396  cl_now = cl_next;
397  #pragma omp parallel sections
398  {
399  #pragma omp section
400  {
401  write_clustered_trajectory(stringprintf(remapped_name, d), cl_now);
402  }
403  #pragma omp section
404  {
405  cl_next = read_clustered_trajectory(stringprintf(basename, d + d_step));
406  max_id = *std::max_element(cl_now.begin(), cl_now.end());
407  for (std::size_t i=0; i < n_rows; ++i) {
408  if (cl_next[i] != 0) {
409  cl_next[i] += max_id;
410  if (cl_now[i] != 0) {
411  network[cl_now[i]] = cl_next[i];
412  ++pops[cl_now[i]];
413  free_energies[cl_now[i]] = d;
414  }
415  }
416  }
417  }
418  }
419  }
420  // handle last trajectory
421  Clustering::logger(std::cout) << "free energy level: " << stringprintf("%0.2f", d_max) << std::endl;
422  cl_now = cl_next;
423  write_clustered_trajectory(stringprintf(remapped_name, d_max), cl_now);
424  for (std::size_t i=0; i < n_rows; ++i) {
425  if (cl_now[i] != 0) {
426  ++pops[cl_now[i]];
427  free_energies[cl_now[i]] = d_max;
428  }
429  }
430  // if minpop given: delete nodes and edges not fulfilling min. population criterium
431  if (minpop > 1) {
432  Clustering::logger(std::cout) << "cleaning from low pop. states ..." << std::endl;
433  std::unordered_set<std::size_t> removals;
434  auto pop_it = pops.begin();
435  Clustering::logger(std::cout) << " ... search nodes to remove" << std::endl;
436  while (pop_it != pops.end()) {
437  if (pop_it->second < minpop) {
438  removals.insert(pop_it->first);
439  pops.erase(pop_it++); // as above
440  } else {
441  ++pop_it;
442  }
443  }
444  Clustering::logger(std::cout) << " ... search edges to remove" << std::endl;
445  auto net_it = network.begin();
446  while (net_it != network.end()) {
447  std::size_t a = net_it->first;
448  std::size_t b = net_it->second;
449  if (removals.count(a) > 0 || removals.count(b) > 0) {
450  network.erase(net_it++);
451  } else {
452  ++net_it;
453  }
454  }
455  Clustering::logger(std::cout) << " ... finished." << std::endl;
456  }
457  // save links (i.e. edges) as two-column ascii in a child -> parent manner
458  save_network_links("network_links.dat", network);
459  // save id, population and free energy of nodes
460  save_node_info("network_nodes.dat", free_energies, pops);
461  // compute and directly save network end-nodes (i.e. leaves of the network-tree)
462  std::set<std::size_t> leaves = compute_and_save_leaves("network_leaves.dat", network);
463  // save the trajectory consisting of the 'leaf-states'.
464  // all non-leaf states are kept as non-assignment state '0'.
465  save_traj_of_leaves("network_end_node_traj.dat", leaves, d_min, d_max, d_step, remapped_name, n_rows);
466  // generate html-file with embedded javascript to visualize network
467  save_network_to_html("network_visualization.html", network, free_energies, pops);
468  }
