38  BoxGrid
39  compute_box_grid(const float* coords,
40  const std::size_t n_rows,
41  const std::size_t n_cols,
42  const float radius) {
43  // use first, second and third coordinates, since these usually
44  // correspond to first, second and third PCs, having highest variance.
45  // if clustering is only in 1 or 2 dimensions, boxes of higher dimensions
46  // will be kept empty.
47  const int BOX_DIM_1 = 0;
48  const int BOX_DIM_2 = 1;
49  const int BOX_DIM_3 = 2;
50  BoxGrid grid;
51  ASSUME_ALIGNED(coords);
52  // find min/max values for first and second dimension
53  float min_x1=0.0f, max_x1=0.0f, min_x2=0.0f, max_x2=0.0f, min_x3=0.0f, max_x3=0.0f;
54  min_x1=coords[0*n_cols+BOX_DIM_1];
55  max_x1=coords[0*n_cols+BOX_DIM_1];
56  if (n_cols > 1) {
57  min_x2=coords[0*n_cols+BOX_DIM_2];
58  max_x2=coords[0*n_cols+BOX_DIM_2];
59  }
60  if (n_cols > 2) {
61  min_x3=coords[0*n_cols+BOX_DIM_3];
62  max_x3=coords[0*n_cols+BOX_DIM_3];
63  }
64  Clustering::logger(std::cout) << "setting up boxes for fast NN search" << std::endl;
65  for (std::size_t i=1; i < n_rows; ++i) {
66  min_x1 = std::min(min_x1, coords[i*n_cols+BOX_DIM_1]);
67  max_x1 = std::max(max_x1, coords[i*n_cols+BOX_DIM_1]);
68  if (n_cols > 1) {
69  min_x2 = std::min(min_x2, coords[i*n_cols+BOX_DIM_2]);
70  max_x2 = std::max(max_x2, coords[i*n_cols+BOX_DIM_2]);
71  }
72  if (n_cols > 2) {
73  min_x3 = std::min(min_x3, coords[i*n_cols+BOX_DIM_3]);
74  max_x3 = std::max(max_x3, coords[i*n_cols+BOX_DIM_3]);
75  }
76  }
77  // build 3D grid with boxes for efficient nearest neighbor search
78  grid.n_boxes.push_back(fabs(max_x1 - min_x1) / radius + 1);
79  if (n_cols > 1) {
80  grid.n_boxes.push_back(fabs(max_x2 - min_x2) / radius + 1);
81  }
82  if (n_cols > 2) {
83  grid.n_boxes.push_back(fabs(max_x3 - min_x3) / radius + 1);
84  }
85  grid.assigned_box.resize(n_rows);
86  int i_box_1=0, i_box_2=0, i_box_3=0;
87  for (std::size_t i=0; i < n_rows; ++i) {
88  i_box_1 = (coords[i*n_cols+BOX_DIM_1] - min_x1) / radius;
89  if (n_cols > 1) {
90  i_box_2 = (coords[i*n_cols+BOX_DIM_2] - min_x2) / radius;
91  }
92  if (n_cols > 2) {
93  i_box_3 = (coords[i*n_cols+BOX_DIM_3] - min_x3) / radius;
94  }
95  grid.assigned_box[i] = {i_box_1, i_box_2, i_box_3};
96  grid.boxes[grid.assigned_box[i]].push_back(i);
97  }
98  return grid;
99  }

109  is_valid_box(const Box box, const BoxGrid& grid) {
110  int i1 = box[0];
111  int i2 = box[1];
112  int i3 = box[2];
113  return ((i1 >= 0)

115  && (i2 >= 0)

135  std::map<float, std::vector<std::size_t>>
136  calculate_populations(const float* coords,
137  const std::size_t n_rows,
138  const std::size_t n_cols,
139  std::vector<float> radii) {
140  std::map<float, std::vector<std::size_t>> pops;
141  for (float rad: radii) {
142  pops[rad].resize(n_rows, 1);
143  }
144  std::sort(radii.begin(), radii.end(), std::greater<float>());
145  std::size_t n_radii = radii.size();
146  std::vector<float> rad2(n_radii);
147  for (std::size_t i=0; i < n_radii; ++i) {
148  rad2[i] = radii[i]*radii[i];
149  }
150  ASSUME_ALIGNED(coords);
151  std::size_t i, j, k, l, ib;
152  BoxGrid grid = compute_box_grid(coords, n_rows, n_cols, radii[0]);
153  Clustering::logger(std::cout) << " box grid: "
154  << grid.n_boxes[0]
155  << " x "
156  << grid.n_boxes[1]
157  << " x "
158  << grid.n_boxes[2]
159  << std::endl;
160  Clustering::logger(std::cout) << "computing pops" << std::endl;
161  float dist, c;
162  Box box;
163  Box center;
164  int i_neighbor;
165  std::vector<int> box_buffer;
166  #pragma omp parallel for default(none) private(i,box,box_buffer,center,i_neighbor,ib,dist,j,k,l,c) \
167  firstprivate(n_rows,n_cols,n_radii,radii,rad2,N_NEIGHBOR_BOXES) \
168  shared(coords,pops,grid) \
169  schedule(dynamic,1024)
170  for (i=0; i < n_rows; ++i) {
171  center = grid.assigned_box[i];
172  // loop over surrounding boxes to find neighbor candidates
173  for (i_neighbor=0; i_neighbor < N_NEIGHBOR_BOXES; ++i_neighbor) {
174  box = neighbor_box(center, i_neighbor);
175  if (is_valid_box(box, grid)) {
176  box_buffer = grid.boxes[box];
177  // loop over frames inside surrounding box
178  for (ib=0; ib < box_buffer.size(); ++ib) {
179  j = box_buffer[ib];
180  if (i < j) {
181  dist = 0.0f;
182  #pragma simd reduction(+:dist)
183  for (k=0; k < n_cols; ++k) {
184  c = coords[i*n_cols+k] - coords[j*n_cols+k];
185  dist += c*c;
186  }
187  for (l=0; l < n_radii; ++l) {
188  if (dist < rad2[l]) {
189  #pragma omp atomic
190  pops[radii[l]][i] += 1;
191  #pragma omp atomic
192  pops[radii[l]][j] += 1;
193  } else {
194  // if it's not in the bigger radius,
195  // it won't be in the smaller ones.
196  break;
197  }
198  }
199  }
200  }
201  }
202  }
203  }
204  return pops;
205  }

207  std::vector<float>
208  calculate_free_energies(const std::vector<std::size_t>& pops) {
209  std::size_t i;
210  const std::size_t n_frames = pops.size();
211  const float max_pop = (float) ( * std::max_element(pops.begin(), pops.end()));
212  std::vector<float> fe(n_frames);
213  #pragma omp parallel for default(none) private(i) firstprivate(max_pop, n_frames) shared(fe, pops)
214  for (i=0; i < n_frames; ++i) {
215  fe[i] = (float) -1 * log(pops[i]/max_pop);
216  }
217  return fe;
218  }

220  std::vector<FreeEnergy>
221  sorted_free_energies(const std::vector<float>& fe) {
222  std::vector<FreeEnergy> fe_sorted;
223  for (std::size_t i=0; i < fe.size(); ++i) {
224  fe_sorted.push_back(FreeEnergy(i, fe[i]));
225  }
226  // sort for free energy: lowest to highest (low free energy = high density)
227  std::sort(fe_sorted.begin(),
228  fe_sorted.end(),
229  [] (const FreeEnergy& d1, const FreeEnergy& d2) -> bool {return d1.second < d2.second;});
230  return fe_sorted;
231  }

233  std::tuple<Neighborhood, Neighborhood>
234  nearest_neighbors(const float* coords,
235  const std::size_t n_rows,
236  const std::size_t n_cols,
237  const std::vector<float>& free_energy) {
238  Neighborhood nh;
239  Neighborhood nh_high_dens;
240  // initialize neighborhood
241  for (std::size_t i=0; i < n_rows; ++i) {
242  nh[i] = Neighbor(n_rows+1, std::numeric_limits<float>::max());
243  nh_high_dens[i] = Neighbor(n_rows+1, std::numeric_limits<float>::max());
244  }
245  // calculate nearest neighbors with distances
246  std::size_t i, j, c, min_j, min_j_high_dens;
247  float dist, d, mindist, mindist_high_dens;
248  ASSUME_ALIGNED(coords);
249  #pragma omp parallel for default(none) \
250  private(i,j,c,dist,d,mindist,mindist_high_dens,min_j,min_j_high_dens) \
251  firstprivate(n_rows,n_cols) \
252  shared(coords,nh,nh_high_dens,free_energy) \
253  schedule(dynamic, 2048)
254  for (i=0; i < n_rows; ++i) {
255  mindist = std::numeric_limits<float>::max();
256  mindist_high_dens = std::numeric_limits<float>::max();
257  min_j = n_rows+1;
258  min_j_high_dens = n_rows+1;
259  for (j=0; j < n_rows; ++j) {
260  if (i != j) {
261  dist = 0.0f;
262  #pragma simd reduction(+:dist)
263  for (c=0; c < n_cols; ++c) {
264  d = coords[i*n_cols+c] - coords[j*n_cols+c];
265  dist += d*d;
266  }
267  // direct neighbor
268  if (dist < mindist) {
269  mindist = dist;
270  min_j = j;
271  }
272  // next neighbor with higher density / lower free energy
273  if (free_energy[j] < free_energy[i] && dist < mindist_high_dens) {
274  mindist_high_dens = dist;
275  min_j_high_dens = j;
276  }
277  }
278  }
279  nh[i] = Neighbor(min_j, mindist);
280  nh_high_dens[i] = Neighbor(min_j_high_dens, mindist_high_dens);
281  }
282  return std::make_tuple(nh, nh_high_dens);
283  }

285  // returns neighborhood set of single frame.
286  // all ids are sorted in free energy.
287  std::set<std::size_t>
288  high_density_neighborhood(const float* coords,
289  const std::size_t n_cols,
290  const std::vector<FreeEnergy>& sorted_fe,
291  const std::size_t i_frame,
292  const std::size_t limit,
293  const float max_dist) {
294  // buffer to hold information whether frame i is
295  // in neighborhood (-> assign 1) or not (-> keep 0)
296  std::vector<int> frame_in_nh(limit, 0);
297  std::set<std::size_t> nh;
298  std::size_t j,c;
299  const std::size_t i_frame_sorted = sorted_fe[i_frame].first * n_cols;
300  float d,dist2;
301  ASSUME_ALIGNED(coords);
302  #pragma omp parallel for default(none) private(j,c,d,dist2) \
303  firstprivate(i_frame,i_frame_sorted,limit,max_dist,n_cols) \
304  shared(coords,sorted_fe,frame_in_nh)
305  for (j=0; j < limit; ++j) {
306  if (i_frame != j) {
307  dist2 = 0.0f;
308  #pragma simd reduction(+:dist2)
309  for (c=0; c < n_cols; ++c) {
310  d = coords[i_frame_sorted+c] - coords[sorted_fe[j].first*n_cols+c];
311  dist2 += d*d;
312  }
313  if (dist2 < max_dist) {
314  frame_in_nh[j] = 1;
315  }
316  }
317  }
318  // reduce buffer data to real neighborhood structure
319  for (j=0; j < limit; ++j) {
320  if (frame_in_nh[j] > 0) {
321  nh.insert(j);
322  }
323  }
324  nh.insert(i_frame);
325  return nh;
326  }

328  double
329  compute_sigma2(const Neighborhood& nh) {
330  double sigma2 = 0.0;
331  for (auto match: nh) {
332  // first second: nearest neighbor info
333  // second second: squared dist to nearest neighbor
334  sigma2 += match.second.second;
335  }
336  return (sigma2 / nh.size());
337  }

339  std::vector<std::size_t>
340  assign_low_density_frames(const std::vector<std::size_t>& initial_clustering,
341  const Neighborhood& nh_high_dens,
342  const std::vector<float>& free_energy) {
343  std::vector<FreeEnergy> fe_sorted = sorted_free_energies(free_energy);
344  std::vector<std::size_t> clustering(initial_clustering);
345  for (const auto& fe: fe_sorted) {
346  std::size_t id = fe.first;
347  if (clustering[id] == 0) {
348  std::size_t neighbor_id = nh_high_dens.find(id)->second.first;
349  // assign cluster of nearest neighbor with higher density
350  clustering[id] = clustering[neighbor_id];
351  }
352  }
353  return clustering;
354  }
