40  WTDMap
41  compute_wtd(std::list<std::size_t> streaks) {
42  WTDMap wtd;
43  if (streaks.size() > 0) {
44  streaks.sort(std::greater<std::size_t>());
45  std::size_t max_streak = streaks.front();
46  for (std::size_t i=0; i <= max_streak; ++i) {
47  float n_steps = 0.0f;
48  for (auto s: streaks) {
49  if (i > s) {
50  break;
51  }
52  n_steps += 1.0f;
53  }
54  wtd[i] = n_steps / ((float) streaks.size());
55  }
56  }
57  return wtd;
58  }

61  main(boost::program_options::variables_map args) {
62  using namespace Clustering::Tools;
63  namespace b_po = boost::program_options;
64  // load states
65  Clustering::logger(std::cout) << "~~~ reading files\n"
66  << " trajectory from: " << args["states"].as<std::string>()
67  << std::endl;
68  std::vector<std::size_t> states = Clustering::Tools::read_clustered_trajectory(args["states"].as<std::string>());
69  std::set<std::size_t> state_names(states.begin(), states.end());
70  std::size_t n_frames = states.size();
71  std::string header_comment = args["header"].as<std::string>();
72  if (args.count("output") || args.count("distribution") || args.count("cores")) {
73  // load concatenation limits to treat concatenated trajectories correctly
74  // when performing dynamical corrections
75  std::vector<std::size_t> concat_limits;
76  if (args.count("concat-limits")) {
77  Clustering::logger(std::cout) << " limits from: "
78  << args["concat-limits"].as<std::string>() << std::endl;
79  concat_limits = Clustering::Tools::read_concat_limits(args["concat-limits"].as<std::string>());
80  } else if (args.count("concat-nframes")) {
81  std::size_t n_frames_per_subtraj = args["concat-nframes"].as<std::size_t>();
82  for (std::size_t i=n_frames_per_subtraj; i <= n_frames; i += n_frames_per_subtraj) {
83  concat_limits.push_back(i);
84  }
85  } else {
86  concat_limits = {n_frames};
87  }
88  // check if concat_limits are well definied
89  Clustering::Tools::check_concat_limits(concat_limits, n_frames);
90  // load window size information
91  Clustering::logger(std::cout) << "\n\n~~~ coring windows:\n from file: "
92  << args["windows"].as<std::string>() << std::endl;
93  std::map<std::size_t, std::size_t> coring_windows;
94  {
95  std::ifstream ifs(args["windows"].as<std::string>());
96 // std::string buf1, buf2;
97  std::string buf;
98  std::size_t state, window;
99  std::size_t size_for_all = 1;
100  if (ifs.fail()) {
101  std::cerr << "error: cannot open file '"
102  << args["windows"].as<std::string>()
103  << "'" << std::endl;
104  exit(EXIT_FAILURE);
105  } else {
106  while (ifs.good()) {
107  if (std::isdigit(ifs.peek())) {
108  ifs >> state;
109  ifs >> window;
110  if ( ! ifs.fail()) {
111  coring_windows[state] = window;
112  } else {
113  std::cerr << "error: file not correctly formated." << std::endl;
114  }
115  } else if (ifs.peek() == '*') {
116  ifs >> buf;
117  ifs >> window;
118  if ( ! ifs.fail()) {
119  size_for_all = window;
120  } else {
121  std::cerr << "error: file not correctly formated." << std::endl;
122  }
123  } else {
124  ifs.ignore(std::numeric_limits<std::streamsize>::max(), '\n');
125  }
126  }
127 // while (ifs.good()) {
128 // ifs >> buf1;
129 // ifs >> buf2;
130 // if (ifs.good()) {
131 // if (buf1 == "*") {
132 // size_for_all = string_to_num<std::size_t>(buf2);
133 // } else {
134 // coring_windows[string_to_num<std::size_t>(buf1)] = string_to_num<std::size_t>(buf2);
135 // }
136 // }
137 // }
138  }
139  // fill remaining, not explicitly defined states with common window size
140  std::size_t undefined_windows = 0;
141  for (std::size_t name: state_names) {
142  if ( ! coring_windows.count(name)){
143  coring_windows[name] = size_for_all;
144  ++undefined_windows;
145  }
146  }
147  header_comment.append(Clustering::Tools::stringprintf(
148  "#\n# coring specific parameters: \n"
149  "# %i state-specific coring windows were read\n"
150  "# %i frames is used for reamining states\n",
151  state_names.size() - undefined_windows, size_for_all));
152  Clustering::logger(std::cout) << " " << state_names.size() - undefined_windows
153  << " state-specific coring windows were read" << std::endl;
154  if (size_for_all > 1) {
155  Clustering::logger(std::cout) << " default window was set to " << size_for_all
156  << " frames" << std::endl;
157  }
158  }
159  // core trajectory
160  Clustering::logger(std::cout) << "\n~~~ coring trajectory" << std::endl;
161  std::vector<std::size_t> cored_traj(n_frames);
162  std::size_t current_core = states[0];
163  std::vector<long> cores(n_frames);
164  std::size_t changed_frames = 0;
165  // honour concatenation limits, i.e. treat every concatenated trajectory-part on its own
166  std::size_t last_limit = 0;
167  for (std::size_t next_limit: concat_limits) {
168  current_core = states[last_limit];
169  std::size_t next_limit_corrected = std::min(next_limit, n_frames);
170  for (std::size_t i=last_limit; i < next_limit_corrected; ++i) {
171  // coring window
172  std::size_t w = std::min(i+coring_windows[states[i]], next_limit);
173  bool is_in_core = true;
174  for (std::size_t j=i+1; j < w; ++j) {
175  if (states[j] != states[i]) {
176  is_in_core = false;
177  break;
178  }
179  }
180  if (is_in_core) {
181  current_core = states[i];
182  cores[i] = current_core;
183  } else {
184  ++changed_frames;
185  cores[i] = -1;
186  }
187  cored_traj[i] = current_core;
188  }
189  last_limit = next_limit_corrected;
190  }
191  float changed_frames_perc = (float) 100*changed_frames / n_frames;
192  Clustering::logger(std::cout) << Clustering::Tools::stringprintf(" %.2f", changed_frames_perc)
193  << "% of frames were changed\n " << changed_frames
194  << " frames in total"
195  << "\n store result in: " << args["output"].as<std::string>()
196  << std::endl;
197  // write cored trajectory to file
198  std::string header_coring = header_comment
199  + Clustering::Tools::stringprintf("# %.2f", changed_frames_perc)
200  + "% of frames were changed\n";
201  if (args.count("output")) {
202  Clustering::Tools::write_clustered_trajectory(args["output"].as<std::string>(),
203  cored_traj,
204  header_coring);
205  }
206  // write core information to file
207  if (args.count("cores")) {
208  Clustering::Tools::write_single_column<long>(args["cores"].as<std::string>(),
209  cores,
210  header_coring,
211  false);
212  }
213  // compute/save escape time distributions
214  if (args.count("distribution")) {
215  Clustering::logger(std::cout) << "~~~ generating distribution" << std::endl;
216  std::map<std::size_t, std::list<std::size_t>> streaks;
217  std::size_t current_state = cored_traj[0];
218  long n_counts = 0;
219  for (std::size_t state: cored_traj) {
220  if (state == current_state) {
221  ++n_counts;
222  } else {
223  streaks[current_state].push_back(n_counts);
224  current_state = state;
225  n_counts = 1;
226  }
227  }
228  streaks[current_state].push_back(n_counts);
229 
230  std::map<std::size_t, WTDMap> etds;
231  for (std::size_t state: state_names) {
232  etds[state] = compute_wtd(streaks[state]);
233  }
234  // write WTDs to file
235  Clustering::logger(std::cout) << " storing..." << std::endl;
236  for (auto state_etd: etds) {
237  std::string fname = Clustering::Tools::stringprintf(args["distribution"].as<std::string>() + "_%d", state_etd.first);
238  Clustering::Tools::write_map<std::size_t, float>(fname, state_etd.second, header_comment);
239  }
240  }
241  } else {
242  std::cerr << "\n" << "error (coring): nothing to do! please define '--output', '--distribution' or both!" << "\n\n";
243  exit(EXIT_FAILURE);
244  }
245  }
