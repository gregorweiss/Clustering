41  void
42  main(boost::program_options::variables_map args) {
43  using namespace Clustering::Tools;
44  namespace b_fs = boost::filesystem;
45  namespace b_po = boost::program_options;
46  // load states
47  Clustering::logger(std::cout) << "~~~ reading files\n"
48  << " trajectory from: " << args["states"].as<std::string>()
49  << std::endl;
50  std::vector<std::size_t> states = Clustering::Tools::read_clustered_trajectory(args["states"].as<std::string>());
51  std::vector<std::size_t> states_without_noise;
52  std::copy(states.begin(), states.end(), std::back_inserter(states_without_noise));
53  std::set<std::size_t> state_names(states.begin(), states.end());
54  std::size_t n_frames = states.size();
55  // shift cmin fomr [0,100] -> [0,1]
56  float cmin = 0.01*args["cmin"].as<float>();
57  std::string basename = args["basename"].as<std::string>();
58  Clustering::verbose = args["verbose"].as<bool>();
59  // generate header comment
60  std::string header_comment = args["header"].as<std::string>();
61  // noise state is 1 lower than lowest
62  auto lowestState = std::min_element(states.begin(), states.end());
63  std::size_t noiseState = *lowestState-1;
64 
65  if (args.count("output") || args.count("cores")) {
66  // load concatenation limits to treat concatenated trajectories correctly
67  // when performing dynamical corrections
68  std::vector<std::size_t> concat_limits;
69  if (args.count("concat-limits")) {
70  Clustering::logger(std::cout) << " limits from: "
71  << args["concat-limits"].as<std::string>() << std::endl;
72  concat_limits = Clustering::Tools::read_concat_limits(args["concat-limits"].as<std::string>());
73  } else if (args.count("concat-nframes")) {
74  std::size_t n_frames_per_subtraj = args["concat-nframes"].as<std::size_t>();
75  for (std::size_t i=n_frames_per_subtraj; i <= n_frames; i += n_frames_per_subtraj) {
76  concat_limits.push_back(i);
77  }
78  } else {
79  concat_limits = {n_frames};
80  }
81  // check if concat_limits are well definied
82  Clustering::Tools::check_concat_limits(concat_limits, n_frames);
83  // findest highest clust file
84  b_fs::path cwd(b_fs::current_path());
85 
86  typedef std::vector<b_fs::path> vec; // store paths,
87  vec v; // so we can sort them later
88 
89  std::copy(b_fs::directory_iterator(cwd), b_fs::directory_iterator(), std::back_inserter(v));
90 
91  std::sort(v.begin(), v.end()); // sort, since directory iteration
92  // is not ordered on some file systems
93 // std::cout << " basename: " << basename << "\n";
94 // std::cout << " v.end(): " << v.back().string() << "\n";
95  // iterate reverse
96  std::string clust_filename;
97  std::size_t found;
98  for (vec::reverse_iterator it(v.rbegin()), it_end(v.rend()); it != it_end; ++it)
99  {
100  std::ostringstream oss;
101  oss << *it;
102  std::string file_str = oss.str();
103 
104  found = file_str.rfind(basename);
105  if (found!=std::string::npos) {
106  clust_filename = file_str.substr(found,file_str.length()-found-1);
107  break;
108  }
109  }
110  // catch if file not found
111  if (found == std::string::npos) {
112  std::cerr << "\n" << "error (noise): cluster file of type " << basename << " not found\n\n";
113  exit(EXIT_FAILURE);
114  }
115 
116  // open highest clust file
117  header_comment.append(Clustering::Tools::stringprintf("#\n# Execution remarks:\n"
118  "# used for highest cluster file: %s\n", clust_filename.c_str()));
119  Clustering::logger(std::cout) << " highest cluster: " << clust_filename << std::endl;
120  std::vector<std::size_t> clust = Clustering::Tools::read_clustered_trajectory(clust_filename);
121  if (n_frames != clust.size()) {
122  std::cerr << "\n" << "error (noise): clust file is not of same length as state trajectory." << "\n\n";
123  exit(EXIT_FAILURE);
124  }
125  // generate counts of each cluster id
126  CounterClustMap counts;
127  for (std::size_t i=0; i < n_frames; ++i) {
128  CounterClustMap::iterator it(counts.find(clust[i]));
129  if (it != counts.end()){
130  it->second++;
131  } else {
132  counts[clust[i]] = 1;
133  }
134  }
135  Clustering::logger(std::cout) << "~~~ noise assignment" << std::endl;
136  std::size_t noise_frames = 0;
137  // define noise frames as state 0
138  for (std::size_t i=0; i < n_frames; ++i) {
139  if (counts[clust[i]] < cmin*n_frames){
140  states[i] = noiseState;
141  ++noise_frames;
142  }
143  }
144  float noise_frames_per = (float) (100*noise_frames / n_frames);
145  Clustering::logger(std::cout) << Clustering::Tools::stringprintf(" %.2f", noise_frames_per)
146  << "% of frames were identified as noise" << std::endl;
147  header_comment.append(Clustering::Tools::stringprintf(
148  " %.2f% of frames were identified as noise", noise_frames_per));
149  // TODO: remove following line. Should we keep with argument?
150  // Clustering::Tools::write_clustered_trajectory("microstatesNoiseDef", states);
151  // noise core trajectory
152  std::vector<std::size_t> noise_traj(n_frames);
153  std::size_t current_core = states[0];
154  std::vector<long> cores(n_frames);
155  std::size_t changed_frames = 0;
156  // honour concatenation limits, i.e. treat every concatenated trajectory-part on its own
157  std::size_t last_limit = 0;
158  for (std::size_t next_limit: concat_limits) {
159  std::size_t next_limit_corrected = std::min(next_limit, n_frames);
160  for (std::size_t i=last_limit; i < next_limit_corrected; ++i) {
161  if (states[i] != noiseState) {
162  current_core = states[i];
163  break;
164  }
165  }
166  for (std::size_t i=last_limit; i < next_limit_corrected; ++i) {
167  // coring window
168  if (states[i] != noiseState) {
169  current_core = states[i];
170  cores[i] = current_core;
171  } else {
172  cores[i] = -1;
173  }
174  if (current_core != states_without_noise[i]) {
175  ++changed_frames;
176  }
177  noise_traj[i] = current_core;
178  }
179  last_limit = next_limit_corrected;
180  }
181  Clustering::logger(std::cout) << Clustering::Tools::stringprintf(" %.2f", (float) 100*changed_frames / n_frames)
182  << "% of frames were reassigned\n"
183  << " store result in: " << args["output"].as<std::string>()
184  << std::endl;
185  // write cored trajectory to file
186  if (args.count("output")) {
187  Clustering::Tools::write_clustered_trajectory(args["output"].as<std::string>(),
188  noise_traj,
189  header_comment);
190  }
191  // write core information to file
192  if (args.count("cores")) {
193  Clustering::Tools::write_single_column<long>(args["cores"].as<std::string>(),
194  cores,
195  header_comment,
196  false);
197  }
198  } else {
199  std::cerr << "\n" << "error (noise): nothing to do! please define '--output' or '--cores'" << "\n\n";
200  exit(EXIT_FAILURE);
201  }
202  }
203 } // end namespace Noise
204 } // end namespace Clustering
